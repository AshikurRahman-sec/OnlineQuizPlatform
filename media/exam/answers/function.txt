.MODEL SMALL
.STACK 100H
.DATA 
TOTAL DW ?
.CODE
   MAIN PROC
    MOV AX,@DATA
    MOV DS, AX 
    MOV TOTAL,0
    MOV AX,0
    MOV BX,0
    MOV CX,0
    CALL INPUT
            ;TOTAL=TOTAL*10+DIGIT 
    
    
    MOV AH,2
    MOV DL,0DH
    INT 21H
    MOV DL,0AH
    INT 21H 
    
    
    
    MOV CX, 0        
    MOV AX,0
    MOV AX,TOTAL
    
    

    
    CALL OUTPUT
    MOV AH,4CH
    INT 21H

    
    MAIN ENDP

    

    
    INPUT PROC 
        
         INPUTLOOP: 
         
                    MOV DX,10
                    MOV AH,1
                    INT 21H 
                    CMP AL,0DH
                    JE  RETURN 
                    INC CX
                    SUB AL,'0'
                    MOV AH,0
                    MOV BL,AL
                    MOV AX,TOTAL 
                    MUL DX
                    ADD AX,BX 
                    MOV TOTAL,AX
                    JMP INPUTLOOP

                    
             RETURN: RET
                    INPUT ENDP 

      

      
      OUTPUT PROC
         OUTPUTLOOP:
         
         MOV BL,10
         DIV BL      
         MOV DL,AH
         PUSH DX
         MOV AH,0 
         INC CX
         CMP AL,0
         JE PRINT
         
         JMP OUTPUTLOOP

         
         PRINT:
          
         POP DX
         ADD DL,'0'
         MOV AH,2 
         INT 21H
         DEC CX
         CMP CX,0
         JNE PRINT 

         

                
             RET
         OUTPUT ENDP
